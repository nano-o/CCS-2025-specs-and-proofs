Theory AxiomaticDomainModel

theory AxiomaticDomainModel
  imports AbstractDomainModel
begin

text ‹In this theory we define types of quorums corresponding to the various thresholds
used by the optimistic RBC algorithm. This will then allow us to do proof abstractly, just relying
on intersection properties and not on cardinalitites.›

section ‹The domain model, axiomatically›

typedecl party

axiomatization faulty :: "party set" and broadcaster :: party
  where party_finite: "finite (UNIV :: party set)"

instantiation party :: finite
begin

instance
  by intro_classes (simp add: party_finite)

end

definition n :: nat where
  "n ≡ card (UNIV :: party set)"

definition f :: nat where
  "f ≡ card faulty"

axiomatization where
  fault_bound: "3*f < n"
  and two_parties: "2 ≤ n"

typedef opt_quorum =
  "{S :: party set. broadcaster ∉ S ∧ card S >= max 1 ⌈real (n + 2 * f - 2) / 2⌉}"
proof -
  text ‹Proof sketch: @{term "UNIV - {broadcaster}"} excludes @{term broadcaster} and has
    cardinality @{term "n - 1"}. The global inequalities @{thm [source] fault_bound} and
    @{thm [source] two_parties} make this large enough for the optimistic threshold and ensure
    that it is nonempty.›
  let ?S = "UNIV - {broadcaster} :: party set"
  have card_S: "card ?S = n - 1"
    by (simp add: n_def)
  have ge_one: "card ?S >= 1"
  proof -
    have "n - 1 >= 1"
      using two_parties by linarith
    then show ?thesis
      by (simp add: card_S)
  qed
  have thresh: "card ?S >= ⌈real (n + 2 * f - 2) / 2⌉"
  proof -
    have "real (n - 1) >= real (n + 2 * f - 2) / 2"
      using fault_bound by linarith
    then have "⌈real (n - 1)⌉ >= ⌈real (n + 2 * f - 2) / 2⌉"
      by (intro ceiling_mono)
    then have "n - 1 >= ⌈real (n + 2 * f - 2) / 2⌉"
      by simp
    then show ?thesis
      by (simp add: card_S)
  qed
  show "∃S :: party set.
      S ∈ {S :: party set. broadcaster ∉ S ∧ card S >= max 1 ⌈real (n + 2 * f - 2) / 2⌉}"
    using ge_one thresh by (intro exI[of _ ?S]) simp
qed

typedef maj_quorum =
  "{S :: party set. broadcaster ∉ S ∧ card S >= ⌈real n / 2⌉}"
proof -
  text ‹Proof sketch: @{term "UNIV - {broadcaster}"} again provides all non-broadcaster parties.
    The global inequality @{term "2 ≤ n"} ensures that this set reaches the vote threshold.›
  let ?S = "UNIV - {broadcaster} :: party set"
  have card_S: "card ?S = n - 1"
    by (simp add: n_def)
  have thresh: "card ?S >= ⌈real n / 2⌉"
  proof -
    have "real (n - 1) >= real n / 2"
      using two_parties by linarith
    then have "⌈real (n - 1)⌉ >= ⌈real n / 2⌉"
      by (intro ceiling_mono)
    then have "n - 1 >= ⌈real n / 2⌉"
      by simp
    then show ?thesis
      by (simp add: card_S)
  qed
  show "∃S :: party set.
      S ∈ {S :: party set. broadcaster ∉ S ∧ card S >= ⌈real n / 2⌉}"
    using thresh by (intro exI[of _ ?S]) simp
qed

typedef classic_quorum =
  "{S :: party set. broadcaster ∉ S ∧ card S >= ⌈real (n + f - 1) / 2⌉}"
proof -
  text ‹Proof sketch: use @{term "UNIV - {broadcaster}"} once more. The global inequalities
    @{term fault_bound} and @{term "2 ≤ n"} imply that @{term "n - 1"} dominates the quorum
    threshold.›
  let ?S = "UNIV - {broadcaster} :: party set"
  have card_S: "card ?S = n - 1"
    by (simp add: n_def)
  have thresh: "card ?S >= ⌈real (n + f - 1) / 2⌉"
  proof -
    have "real (n - 1) >= real (n + f - 1) / 2"
      using fault_bound two_parties by linarith
    then have "⌈real (n - 1)⌉ >= ⌈real (n + f - 1) / 2⌉"
      by (intro ceiling_mono)
    then have "n - 1 >= ⌈real (n + f - 1) / 2⌉"
      by simp
    then show ?thesis
      by (simp add: card_S)
  qed
  show "∃S :: party set.
      S ∈ {S :: party set. broadcaster ∉ S ∧ card S >= ⌈real (n + f - 1) / 2⌉}"
    using thresh by (intro exI[of _ ?S]) simp
qed

typedef amplification_quorum =
  "{S :: party set. card S >= f + 1}"
proof -
  text ‹Proof sketch: the full universe has cardinality @{term n}, and the global inequality
    @{thm [source] fault_bound} implies @{prop "n >= f + 1"}.›
  have thresh: "card (UNIV :: party set) >= f + 1"
  proof -
    have "n >= f + 1"
      using fault_bound by linarith
    then show ?thesis
      by (simp add: n_def)
  qed
  show "∃S :: party set. S ∈ {S :: party set. card S >= f + 1}"
    using thresh by (intro exI[of _ "UNIV :: party set"]) simp
qed

typedef commit_quorum =
  "{S :: party set. card S >= 2 * f + 1}"
proof -
  text ‹Proof sketch: the full universe again witnesses nonemptiness because the global inequality
    @{thm [source] fault_bound} implies @{term "n >= 2 * f + 1"}.›
  have thresh: "card (UNIV :: party set) >= 2 * f + 1"
  proof -
    have "n >= 2 * f + 1"
      using fault_bound by linarith
    then show ?thesis
      by (simp add: n_def)
  qed
  show "∃S :: party set. S ∈ {S :: party set. card S >= 2 * f + 1}"
    using thresh by (intro exI[of _ "UNIV :: party set"]) simp
qed

section ‹Intersection properties›

lemma card_lemma_1:
  assumes "broadcaster ∈ faulty"
    and "broadcaster ∉ S⇩1"
    and "card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉"
    and "broadcaster ∉ S⇩2"
    and "card S⇩2 ≥ ⌈real n / 2⌉"
  shows "(S⇩1 ∩ S⇩2) - faulty ≠ {}"
  text ‹Proof sketch: Since broadcaster is not in @{term S⇩1} or @{term S⇩2}, both are subsets of
    @{term "UNIV - {broadcaster}"} which has cardinality @{term "n - 1::nat"}. By inclusion-exclusion,
    @{term "card (S⇩1 ∩ S⇩2)"} is at least @{term "f::nat"}.
    Since broadcaster is in faulty but not in @{term "S⇩1 ∩ S⇩2"}, at most @{term "f - 1::nat"} faulty
    elements can be in the intersection. So @{term "(S⇩1 ∩ S⇩2) - faulty"} has at least 1 element.›
proof -
  have n_pos: "n ≥ 1" using fault_bound by (simp add: n_def)
  have f_pos: "f ≥ 1"
    by (metis One_nat_def ‹broadcaster ∈ faulty› bot_nat_0.extremum_unique card.remove f_def finite nat.simps(3) not_less_eq_eq)
  have card_U: "card (UNIV - {broadcaster} :: party set) = n - 1"
    using n_pos by (simp add: n_def)
  text ‹By inclusion-exclusion, the intersection is large enough.›
  have ie: "card (S⇩1 ∩ S⇩2) + (n - 1) ≥ card S⇩1 + card S⇩2"
    by (metis Diff_empty Un_iff add.commute add_le_cancel_right ‹broadcaster ∉ S⇩1› ‹broadcaster ∉ S⇩2› card_U card_Un_Int card_mono finite subset_Diff_insert subset_UNIV)
  text ‹The sum of cardinalities is at least @{term "n + f - 1::nat"} in int.›
  have sum_bound: "int (card S⇩1) + int (card S⇩2) ≥ int n + int f - 1"
  proof -
    have "n + 2*f ≥ 2" using n_pos f_pos by linarith
    hence "real (n + 2*f - 2) / 2 + real n / 2 = real n + real f - 1"
      by (simp add: field_simps)
    thus ?thesis
      using ‹card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉› ‹card S⇩2 ≥ ⌈real n / 2⌉›
        le_of_int_ceiling[of "real (n + 2*f - 2) / 2"]
        le_of_int_ceiling[of "real n / 2"]
      by linarith
  qed
  text ‹So the intersection has at least @{term "f::nat"} elements.›
  have inter_card: "int (card (S⇩1 ∩ S⇩2)) ≥ int f"
    using ie sum_bound n_pos by linarith
  text ‹Broadcaster is in faulty but not in the intersection, so at most @{term "f - 1::nat"}
    faulty elements are in the intersection.›
  have faulty_inter: "card (faulty ∩ (S⇩1 ∩ S⇩2)) ≤ f - 1"
    by (metis Int_iff Int_lower1 One_nat_def ‹broadcaster ∈ faulty› ‹broadcaster ∉ S⇩1› bot_nat_0.extremum_unique card.infinite card.remove card_Diff_singleton_if card_seteq f_def f_pos not_less_eq_eq)
  text ‹Conclude that the intersection minus faulty is nonempty.›
  have "card ((S⇩1 ∩ S⇩2) - faulty) ≥ 1"
    by (metis Diff_Diff_Int Diff_empty Int_lower2 One_nat_def card.empty card_seteq dec_greater_eq_self_imp_bot empty_subsetI f_def f_pos faulty_inter finite inter_card not_less_eq_eq of_nat_le_iff)
  thus ?thesis by (metis card.empty not_one_le_zero)
qed

lemma card_lemma_2:
  ― ‹This follows from @{thm [source] card_lemma_1} because @{term "⌈real (n + f - 1) / 2⌉ ≥ ⌈real n / 2⌉"} when @{term "f>0"}.›
  assumes "broadcaster ∈ faulty"
    and "broadcaster ∉ S⇩1"
    and "card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉"
    and "broadcaster ∉ S⇩2"
    and "card S⇩2 ≥ ⌈real (n + f - 1) / 2⌉"
  shows "(S⇩1 ∩ S⇩2) - faulty ≠ {}"
  text ‹Since @{term "f ≥ 1"}, we have @{term "⌈real (n + f - 1) / 2⌉ ≥ ⌈real n / 2⌉"},
    so the assumption on @{term S⇩2} is at least as strong as what @{thm [source] card_lemma_1} requires.
    We apply @{thm [source] card_lemma_1} directly.›
proof -
  have "f ≥ 1"
    by (metis One_nat_def ‹broadcaster ∈ faulty› bot_nat_0.extremum_unique card.remove f_def finite nat.simps(3) not_less_eq_eq)
  hence "⌈real (n + f - 1) / 2⌉ ≥ ⌈real n / 2⌉"
    by (intro ceiling_mono divide_right_mono) linarith+
  hence "card S⇩2 ≥ ⌈real n / 2⌉"
    using ‹card S⇩2 ≥ ⌈real (n + f - 1) / 2⌉› by linarith
  thus ?thesis
    using card_lemma_1[OF ‹broadcaster ∈ faulty› ‹broadcaster ∉ S⇩1› ‹card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉› ‹broadcaster ∉ S⇩2›]
    by blast
qed

lemma card_lemma_3:
  ― ‹This should follows immediately from @{thm [source] card_lemma_1} because @{term "⌈real (n + 2*f - 2) / 2⌉ ≥ ⌈real n / 2⌉"} when @{term "f>0"}›
  assumes "broadcaster ∈ faulty"
    and "broadcaster ∉ S⇩1" 
    and "card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉"
    and "broadcaster ∉ S⇩2"
    and "card S⇩2 ≥ ⌈real (n + 2*f - 2) / 2⌉"
  shows "((S⇩1 ∩ S⇩2) - faulty) ≠ {}"
  text ‹Since @{term "f ≥ 1"}, we have @{term "⌈real (n + 2*f - 2) / 2⌉ ≥ ⌈real n / 2⌉"},
    so the assumption on @{term S⇩2} implies the weaker bound needed by @{thm [source] card_lemma_1}.
    We apply @{thm [source] card_lemma_1} directly.›
proof -
  have "f ≥ 1"
    by (metis One_nat_def ‹broadcaster ∈ faulty› bot_nat_0.extremum_unique card.remove f_def finite nat.simps(3) not_less_eq_eq)
  hence "⌈real (n + 2*f - 2) / 2⌉ ≥ ⌈real n / 2⌉"
    by (intro ceiling_mono divide_right_mono) linarith+
  hence "card S⇩2 ≥ ⌈real n / 2⌉"
    using ‹card S⇩2 ≥ ⌈real (n + 2*f - 2) / 2⌉› by linarith
  thus ?thesis
    using card_lemma_1[OF ‹broadcaster ∈ faulty› ‹broadcaster ∉ S⇩1› ‹card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉› ‹broadcaster ∉ S⇩2›]
    by blast
qed

lemma card_lemma_4:
  assumes "card S⇩1 ≥ 2*f +1"
  shows "card (S⇩1 - faulty) ≥ f +1"
  text ‹At most @{term f} elements of @{term S⇩1} can be faulty, so removing faulty elements
    leaves at least @{term "2*f + 1 - f"} = @{term "f + (1::nat)"} elements.›
proof -
  have "card (S⇩1 ∩ faulty) ≤ f"
    by (metis Int_lower2 card_mono f_def finite)
  moreover have "card (S⇩1 - faulty) = card S⇩1 - card (S⇩1 ∩ faulty)"
    by (metis card_Diff_subset_Int finite)
  ultimately show ?thesis using assms by linarith
qed

lemma card_lemma_5:
  assumes "broadcaster ∈ faulty"
    and "broadcaster ∉ S⇩1" 
    and "card S⇩1 ≥ ⌈real (n + 2*f - 2) / 2⌉"
  shows "card (S⇩1 - faulty) ≥ ⌈real n / 2⌉"
  text ‹Since @{term "broadcaster ∈ faulty"} but @{term "broadcaster ∉ S⇩1"}, at most @{term "f - 1::nat"}
    faulty elements are in @{term S⇩1}. So @{term "card (S⇩1 - faulty)"} is at least
    @{term "⌈real (n + 2*f - 2) / 2⌉ - int (f - 1)"}. Since
    @{term "⌈real (n + 2*f - 2) / 2⌉ = ⌈real n / 2⌉ + int f - 1"}, the result follows.›
proof -
  have f_pos: "f ≥ 1"
    by (metis One_nat_def ‹broadcaster ∈ faulty› bot_nat_0.extremum_unique card.remove f_def finite nat.simps(3) not_less_eq_eq)
  have "card (S⇩1 ∩ faulty) ≤ f - 1"
    by (metis Int_iff Int_lower2 assms(1,2) card.remove card_Diff_singleton_if card_seteq f_def finite not_less_eq_eq)
  moreover have "card (S⇩1 - faulty) = card S⇩1 - card (S⇩1 ∩ faulty)"
    by (metis card_Diff_subset_Int finite)
  moreover have "⌈real (n + 2*f - 2) / 2⌉ = ⌈real n / 2⌉ + int f - 1"
  proof -
    have "n + 2*f ≥ 2" using f_pos by linarith
    hence "real (n + 2*f - 2) / 2 = real n / 2 + real f - 1"
      by (simp add: field_simps)
    thus ?thesis by linarith
  qed
  moreover have "card (S⇩1 ∩ faulty) ≤ card S⇩1"
    by (simp add: card_mono)
  ultimately have "int (card (S⇩1 - faulty)) ≥ int (card S⇩1) - int (f - 1)"
    by (simp add: card_Diff_subset_Int card_mono)
  moreover have "int (card S⇩1) ≥ ⌈real n / 2⌉ + int f - 1"
    using assms(3) by linarith
  ultimately show ?thesis using f_pos by linarith
qed

lemma card_lemma_6:
  assumes "card S⇩1 ≥ ⌈real n / 2⌉"
  shows "∃ p ∈ S⇩1 . p ∉ faulty"
  text ‹Proof sketch: From @{term "3*f < n"} we get @{term "f < ⌈real n / 2⌉"}, so
    @{term "card faulty < card S⇩1"}. Hence @{term S⇩1} cannot be a subset of faulty,
    meaning there exists a member of @{term S⇩1} not in faulty.›
proof -
  have "int f < ⌈real n / 2⌉"
  proof -
    have "2 * f < n" using fault_bound by linarith
    hence "real (2*f) < real n" by simp
    hence "real f < real n / 2" by simp
    thus ?thesis by linarith
  qed
  hence "int f < int (card S⇩1)" using assms by linarith
  hence "card faulty < card S⇩1" by (simp add: f_def)
  hence "¬ S⇩1 ⊆ faulty"
    by (metis card_mono finite leD)
  thus ?thesis by blast
qed

lemma card_lemma_7:
  assumes "broadcaster ∈ faulty"
    and "broadcaster ∉ S⇩1"
    and "card S⇩1 ≥ ⌈real (n + f - 1) / 2⌉"
    and "broadcaster ∉ S⇩2"
    and "card S⇩2 ≥ ⌈real (n + f - 1) / 2⌉"
  shows "(S⇩1 ∩ S⇩2) - faulty ≠ {}"
  text ‹Proof sketch: Since @{term "broadcaster"} is excluded from @{term S⇩1} and @{term S⇩2},
    both sets lie in @{term "UNIV - {broadcaster}"}, whose cardinality is @{term "n - 1::nat"}.
    Inclusion-exclusion and the lower bounds on @{term "card S⇩1"} and @{term "card S⇩2"}
    yield @{term "card (S⇩1 ∩ S⇩2) ≥ f"}. Because @{term "broadcaster ∈ faulty"} but
    @{term "broadcaster ∉ S⇩1 ∩ S⇩2"}, at most @{term "f - 1::nat"} faulty parties lie
    in the intersection. Therefore @{term "(S⇩1 ∩ S⇩2) - faulty"} is nonempty.›
proof -
  have n_pos: "n ≥ 1"
    using fault_bound by (simp add: n_def)
  have f_pos: "f ≥ 1"
    by (metis One_nat_def ‹broadcaster ∈ faulty› bot_nat_0.extremum_unique card.remove f_def finite nat.simps(3) not_less_eq_eq)
  have card_U: "card (UNIV - {broadcaster} :: party set) = n - 1"
    using n_pos by (simp add: n_def)
  have union_bound: "card (S⇩1 ∪ S⇩2) ≤ n - 1"
    by (metis Diff_empty Un_iff assms(2,4) card_U card_mono finite subset_Diff_insert top_greatest)
  have ie: "card (S⇩1 ∩ S⇩2) + (n - 1) ≥ card S⇩1 + card S⇩2"
    by (metis add.commute add_le_cancel_right card_Un_Int finite union_bound)
  have sum_bound: "int (card S⇩1) + int (card S⇩2) ≥ int n + int f - 1"
  proof -
    have "2 * ⌈real (n + f - 1) / 2⌉ ≥ int n + int f - 1"
      using le_of_int_ceiling[of "real (n + f - 1) / 2"] by linarith
    thus ?thesis
      using ‹card S⇩1 ≥ ⌈real (n + f - 1) / 2⌉› ‹card S⇩2 ≥ ⌈real (n + f - 1) / 2⌉›
      by linarith
  qed
  have inter_card: "card (S⇩1 ∩ S⇩2) ≥ f"
    using ie sum_bound n_pos by linarith
  have faulty_inter: "card (faulty ∩ (S⇩1 ∩ S⇩2)) ≤ f - 1"
    by (metis Int_iff Int_lower1 One_nat_def assms(1,2) card.infinite card.remove
        card_Diff_singleton_if card_seteq dual_order.refl f_def f_pos not_less_eq_eq)
  show ?thesis
    by (metis Diff_eq_empty_iff Int_absorb1 diff_diff_cancel diff_is_0_eq diff_le_mono2 f_pos faulty_inter inter_card not_one_le_zero)
qed

section "Proof that we have an instance of the abstract domain model"

definition opt_quorum_member :: "opt_quorum => party => bool" where
  "opt_quorum_member Q p ≡ p ∈ Rep_opt_quorum Q"

definition maj_quorum_member :: "maj_quorum => party => bool" where
  "maj_quorum_member Q p ≡ p ∈ Rep_maj_quorum Q"

definition classic_quorum_member :: "classic_quorum => party => bool" where
  "classic_quorum_member Q p ≡ p ∈ Rep_classic_quorum Q"

definition amplification_quorum_member :: "amplification_quorum => party => bool" where
  "amplification_quorum_member Q p ≡ p ∈ Rep_amplification_quorum Q"

definition commit_quorum_member :: "commit_quorum => party => bool" where
  "commit_quorum_member Q p ≡ p ∈ Rep_commit_quorum Q"

interpretation axiomatic_abstract_domain_model:
  abstract_domain_model broadcaster faulty opt_quorum_member maj_quorum_member
    classic_quorum_member amplification_quorum_member commit_quorum_member
proof (standard)
  show "¬ classic_quorum_member quQ broadcaster" for quQ
    using Rep_classic_quorum[of quQ] unfolding classic_quorum_member_def by auto

  show "¬ maj_quorum_member vtQ broadcaster" for vtQ
    using Rep_maj_quorum[of vtQ] unfolding maj_quorum_member_def by auto

  show "¬ opt_quorum_member opQ broadcaster" for opQ
    using Rep_opt_quorum[of opQ] unfolding opt_quorum_member_def by auto

  show "∃coQ. ∀p. commit_quorum_member coQ p ⟶ p ∉ faulty"
  text ‹Proof sketch: @{term "(UNIV :: party set) - faulty"} has cardinality @{term "n - f"},
    and @{thm [source] fault_bound} implies @{term "n - f ≥ 2 * f + 1"}. Using this set as
    the carrier of a commit quorum yields a witness whose members are all nonfaulty.›
  proof -
    let ?S = "(UNIV :: party set) - faulty"
    have card_S: "card ?S = n - f"
      by (simp add: n_def f_def Finite_Set.card_Diff_subset)
    have co_carrier: "?S ∈ {S :: party set. card S ≥ 2 * f + 1}"
    proof
      have "n - f ≥ 2 * f + 1"
        using fault_bound by linarith
      with card_S show "card ?S ≥ 2 * f + 1"
        by simp
    qed
    let ?coQ = "Abs_commit_quorum ?S"
    have rep_coQ: "Rep_commit_quorum ?coQ = ?S"
      using co_carrier by (simp add: Abs_commit_quorum_inverse)
    show ?thesis
      using rep_coQ unfolding commit_quorum_member_def by auto
  qed

  show "∃amQ. ∀p. amplification_quorum_member amQ p ⟶ p ∉ faulty"
  text ‹Proof sketch: @{term "(UNIV :: party set) - faulty"} also has enough members for the
    amplification threshold because @{thm [source] fault_bound} implies @{term "n - f ≥ f + 1"}.
    Turning that set into an amplification quorum gives the desired witness.›
  proof -
    let ?S = "(UNIV :: party set) - faulty"
    have card_S: "card ?S = n - f"
      by (simp add: n_def f_def Finite_Set.card_Diff_subset)
    have am_carrier: "?S ∈ {S :: party set. card S ≥ f + 1}"
    proof
      have "n - f ≥ f + 1"
        using fault_bound by linarith
      with card_S show "card ?S ≥ f + 1"
        by simp
    qed
    let ?amQ = "Abs_amplification_quorum ?S"
    have rep_amQ: "Rep_amplification_quorum ?amQ = ?S"
      using am_carrier by (simp add: Abs_amplification_quorum_inverse)
    show ?thesis
      using rep_amQ unfolding amplification_quorum_member_def by auto
  qed

  show "∃vtQ. ∀p. maj_quorum_member vtQ p ⟶ p ∉ faulty"
  text ‹Proof sketch: Use the nonfaulty non-broadcaster parties as the carrier. If the broadcaster
    is faulty this is @{term "UNIV - faulty"}; otherwise it is @{term "(UNIV - faulty) - {broadcaster}"}.
    In both cases the resulting set still reaches the majority threshold.›
  proof -
    let ?S = "((UNIV :: party set) - faulty) - {broadcaster}"
    have card_nonfaulty: "card ((UNIV :: party set) - faulty) = n - f"
      by (simp add: n_def f_def Finite_Set.card_Diff_subset)
    have vt_carrier: "?S ∈ {S :: party set. broadcaster ∉ S ∧ card S ≥ ⌈real n / 2⌉}"
    proof -
      have broadcaster_notin: "broadcaster ∉ ?S"
        by simp
      have card_bound: "card ?S ≥ ⌈real n / 2⌉"

  show "∃p. p ∉ faulty ∧ opt_quorum_member opQ p" for opQ
  text ‹Proof sketch: @{term "Rep_opt_quorum opQ"} is too large to fit inside @{term faulty}.
    The optimistic threshold forces @{term "int f < max 1 ⌈real (n + 2 * f - 2) / 2⌉"}, and the
    representing set has at least that many members, so it must contain a nonfaulty party.›
  proof -
    have opQ_card: "max 1 ⌈real (n + 2 * f - 2) / 2⌉ ≤ int (card (Rep_opt_quorum opQ))"
      using Rep_opt_quorum[of opQ] by auto
    have opt_thresh: "int f < max 1 ⌈real (n + 2 * f - 2) / 2⌉"
    proof (cases "f = 0")
      case True
      then show ?thesis by simp
    next
      case False
      then have f_pos: "f ≥ 1" by simp
      have "4 ≤ n"
        using fault_bound f_pos by linarith
      then have "real f < real (n + 2 * f - 2) / 2"
        by linarith
      then have "int f < ⌈real (n + 2 * f - 2) / 2⌉"
        by (simp add: less_ceiling_iff)
      then show ?thesis
        by simp
    qed
    have "int (card faulty) < int (card (Rep_opt_quorum opQ))"
    proof -

      by (metis card_mono finite leD)
    then obtain p where "p ∈ Rep_opt_quorum opQ" and "p ∉ faulty"

      unfolding opt_quorum_member_def by blast
  qed
